# Magic Perk Points
DI_add_magic_pp = {
	scope = character
	
	effect = {
		random_owned_story = {
			limit = {
				story_type = DI_misc_variables_story
			}
			save_scope_as = DI_story
		}
		
		if = {
			limit = {
				scope:DI_story = { has_variable = DI_pinned_char_var }
			}
			every_living_character = {
				limit = { has_character_flag = pinned_for_edit }

				hidden_effect = {
					add_magic_lifestyle_perk_points = 1
				}
			}
		}
		else = {
			hidden_effect = {
				add_magic_lifestyle_perk_points = 1
			}
		}
	}
}

DI_remove_magic_pp = {
	scope = character
	
	effect = {
		random_owned_story = {
			limit = {
				story_type = DI_misc_variables_story
			}
			save_scope_as = DI_story
		}
		
		if = {
			limit = {
				scope:DI_story = { has_variable = DI_pinned_char_var }
			}
			every_living_character = {
				limit = { has_character_flag = pinned_for_edit }

				hidden_effect = {
					add_magic_lifestyle_perk_points = -1
				}
			}
		}
		else = {
			hidden_effect = {
				add_magic_lifestyle_perk_points = -1
			}
		}
	}
}
